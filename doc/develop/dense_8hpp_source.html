572 template <typename Matrix, typename... TArgs>
573 std::unique_ptr<Matrix> initialize(
574  size_type stride, std::initializer_list<typename Matrix::value_type> vals,
575  std::shared_ptr<const Executor> exec, TArgs &&... create_args)
576 {
577  using dense = matrix::Dense<typename Matrix::value_type>;
578  size_type num_rows = vals.size();
579  auto tmp = dense::create(exec->get_master(), dim<2>{num_rows, 1}, stride);
580  size_type idx = 0;
581  for (const auto &elem : vals) {
582  tmp->at(idx) = elem;
583  ++idx;
584  }
585  auto mtx = Matrix::create(exec, std::forward<TArgs>(create_args)...);
586  tmp->move_to(mtx.get());
587  return mtx;
588 }

642 template <typename Matrix, typename... TArgs>
643 std::unique_ptr<Matrix> initialize(

645  std::initializer_list<std::initializer_list<typename Matrix::value_type>>
646  vals,
647  std::shared_ptr<const Executor> exec, TArgs &&... create_args)
648 {
