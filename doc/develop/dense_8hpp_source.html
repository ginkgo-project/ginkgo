529 template <typename Matrix, typename... TArgs>
530 std::unique_ptr<Matrix> initialize(
531  size_type stride, std::initializer_list<typename Matrix::value_type> vals,
532  std::shared_ptr<const Executor> exec, TArgs &&... create_args)
533 {
534  using dense = matrix::Dense<typename Matrix::value_type>;
535  size_type num_rows = vals.size();
536  auto tmp = dense::create(exec->get_master(), dim<2>{num_rows, 1}, stride);
537  size_type idx = 0;
538  for (const auto &elem : vals) {
539  tmp->at(idx) = elem;
540  ++idx;
541  }
542  auto mtx = Matrix::create(exec, std::forward<TArgs>(create_args)...);
543  tmp->move_to(mtx.get());
544  return mtx;
545 }

599 template <typename Matrix, typename... TArgs>
600 std::unique_ptr<Matrix> initialize(

602  std::initializer_list<std::initializer_list<typename Matrix::value_type>>
603  vals,
604  std::shared_ptr<const Executor> exec, TArgs &&... create_args)
