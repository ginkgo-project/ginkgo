315  std::unique_ptr<Dense> create_submatrix(const span &rows,
316  const span &columns,
317  const size_type stride)
318  {
319  row_major_range range_this{this->get_values(), this->get_size()[0],
320  this->get_size()[1], this->get_stride()};
321  auto range_result = range_this(rows, columns);
322  return Dense::create(
323  this->get_executor(),
324  dim<2>{range_result.length(0), range_result.length(1)},
325  Array<ValueType>::view(
326  this->get_executor(),
327  range_result.length(0) * range_this.length(1),
328  range_result->data),
329  stride);
330  }

441 template <typename Matrix, typename... TArgs>
442 std::unique_ptr<Matrix> initialize(
443  size_type stride, std::initializer_list<typename Matrix::value_type> vals,
444  std::shared_ptr<const Executor> exec, TArgs &&... create_args)
445 {
446  using dense = matrix::Dense<typename Matrix::value_type>;
447  size_type num_rows = vals.size();
448  auto tmp = dense::create(exec->get_master(), dim<2>{num_rows, 1}, stride);
449  size_type idx = 0;
450  for (const auto &elem : vals) {
451  tmp->at(idx) = elem;
452  ++idx;
453  }
454  auto mtx = Matrix::create(exec, std::forward<TArgs>(create_args)...);
455  tmp->move_to(mtx.get());
456  return mtx;
457 }

505 template <typename Matrix, typename... TArgs>
506 std::unique_ptr<Matrix> initialize(
507  size_type stride,
508  std::initializer_list<std::initializer_list<typename Matrix::value_type>>
509  vals,
510  std::shared_ptr<const Executor> exec, TArgs &&... create_args)
511 {
512  using dense = matrix::Dense<typename Matrix::value_type>;
513  size_type num_rows = vals.size();
514  size_type num_cols = num_rows > 0 ? begin(vals)->size() : 1;
515  auto tmp =
516  dense::create(exec->get_master(), dim<2>{num_rows, num_cols}, stride);
517  size_type ridx = 0;
518  for (const auto &row : vals) {
519  size_type cidx = 0;
520  for (const auto &elem : row) {
521  tmp->at(ridx, cidx) = elem;
522  ++cidx;
523  }
524  ++ridx;
525  }
526  auto mtx = Matrix::create(exec, std::forward<TArgs>(create_args)...);
527  tmp->move_to(mtx.get());
