423 template <typename R, typename T>
424 std::unique_ptr<R, std::function<void(R *)>> copy_and_convert_to_impl(
425  std::shared_ptr<const Executor> exec, T *obj)
426 {
427  auto obj_as_r = dynamic_cast<R *>(obj);
428  if (obj_as_r != nullptr && obj->get_executor() == exec) {
429  return {obj_as_r, [](R *) {}};
430  } else {
431  auto copy = R::create(exec);
432  as<ConvertibleTo<xstd::decay_t<R>>>(obj)->convert_to(lend(copy));
433  return {copy.release(), std::default_delete<R>{}};
434  }
435 }
