261 private:
262  static constexpr auto nonpreserving_bits = 4u;
263  static constexpr auto preserving_bits =
264  byte_size * sizeof(storage_type) - nonpreserving_bits;
265  static constexpr auto nonpreserving_mask =
266  storage_type{(0x1 << nonpreserving_bits) - 1};
267  static constexpr auto preserving_mask =
268  storage_type{(0x1 << preserving_bits) - 1} << nonpreserving_bits;
269

289  (preserving << nonpreserving_bits) | nonpreserving))
