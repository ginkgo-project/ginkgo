95  GKO_ATTRIBUTES size_type compute_storage_space(size_type num_blocks) const
96  noexcept

448  static block_interleaved_storage_scheme<index_type> compute_storage_scheme(
449  uint32 max_block_size) noexcept
450  {
451  const auto group_size = static_cast<uint32>(
452  max_block_stride_ / get_superior_power(uint32{2}, max_block_size));
453  const auto block_offset = max_block_size;
454  const auto block_stride = group_size * block_offset;
455  const auto group_offset = max_block_size * block_stride;
456  return {static_cast<index_type>(block_offset),
457  static_cast<index_type>(group_offset),
458  get_significant_bit(group_size)};
459  }
