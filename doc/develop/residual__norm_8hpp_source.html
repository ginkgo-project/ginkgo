133 
134  explicit ResidualNormReduction(const Factory *factory,
135  const CriterionArgs &args)
136  : ResidualNorm<ValueType>(factory->get_executor(),
137  factory->get_parameters().reduction_factor),
138  parameters_{factory->get_parameters()}
139  {
140  if (args.initial_residual == nullptr) {
141  GKO_NOT_SUPPORTED(nullptr);
142  }
143 
144  auto exec = factory->get_executor();
145 
146  auto dense_r = as<Vector>(args.initial_residual);
147  this->starting_tau_ = NormVector::create(
148  exec, dim<2>{1, args.initial_residual->get_size()[1]});
149  this->u_dense_tau_ =
150  NormVector::create_with_config_of(this->starting_tau_.get());
151  dense_r->compute_norm2(this->starting_tau_.get());
152  }
153 };
