205  parameters_{factory->get_parameters()}
206  {
207  auto comp =
208  std::dynamic_pointer_cast<const Composition<value_type>>(lin_op);
209  std::shared_ptr<const LinOp> l_factor;
210  std::shared_ptr<const LinOp> u_factor;
211 
212  // build factorization if we weren't passed a composition
213  if (!comp) {
214  auto exec = lin_op->get_executor();
215  if (!parameters_.factorization_factory) {
216  parameters_.factorization_factory =
217  factorization::ParIlu<value_type, index_type>::build().on(
218  exec);
219  }
220  auto fact = std::shared_ptr<const LinOp>(
221  parameters_.factorization_factory->generate(lin_op));
222  // ensure that the result is a composition
223  comp =
224  std::dynamic_pointer_cast<const Composition<value_type>>(fact);
225  if (!comp) {
226  GKO_NOT_SUPPORTED(comp);
227  }
228  }
229  if (comp->get_operators().size() == 2) {
230  l_factor = comp->get_operators()[0];
231  u_factor = comp->get_operators()[1];

237  auto exec = this->get_executor();

245  if (!parameters_.u_solver_factory) {
