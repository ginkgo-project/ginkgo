167  matrix_data<ValueType, IndexType> get_ordered_data() const
168  {
169  using output_type = matrix_data<ValueType, IndexType>;
170  using nonzero_type = typename output_type::nonzero_type;
171  using entry_type =
172  std::pair<std::pair<index_type, index_type>, value_type>;
173  output_type data{size_};
174  data.nonzeros.reserve(nonzeros_.size());
175  std::transform(nonzeros_.begin(), nonzeros_.end(),
176  std::back_inserter(data.nonzeros), [](entry_type entry) {
177  return nonzero_type{entry.first.first,
