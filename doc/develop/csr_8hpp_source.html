379  void process(const Array<index_type> &mtx_row_ptrs,
380  Array<index_type> *mtx_srow) override
381  {
382  auto nwarps = mtx_srow->get_num_elems();
383 
384  if (nwarps > 0) {
385  auto host_srow_exec = mtx_srow->get_executor()->get_master();
386  auto host_mtx_exec = mtx_row_ptrs.get_executor()->get_master();
387  const bool is_srow_on_host{host_srow_exec ==
388  mtx_srow->get_executor()};
389  const bool is_mtx_on_host{host_mtx_exec ==
390  mtx_row_ptrs.get_executor()};
391  Array<index_type> row_ptrs_host(host_mtx_exec);
392  Array<index_type> srow_host(host_srow_exec);
393  const index_type *row_ptrs{};
394  index_type *srow{};
395  if (is_srow_on_host) {
396  srow = mtx_srow->get_data();
397  } else {
398  srow_host = *mtx_srow;
399  srow = srow_host.get_data();
400  }
401  if (is_mtx_on_host) {
402  row_ptrs = mtx_row_ptrs.get_const_data();
403  } else {
404  row_ptrs_host = mtx_row_ptrs;
405  row_ptrs = row_ptrs_host.get_const_data();
406  }
407  for (size_type i = 0; i < nwarps; i++) {
408  srow[i] = 0;
409  }
410  const auto num_rows = mtx_row_ptrs.get_num_elems() - 1;
411  const auto num_elems = row_ptrs[num_rows];
412  for (size_type i = 0; i < num_rows; i++) {
413  auto bucket =
414  ceildiv((ceildiv(row_ptrs[i + 1], warp_size_) * nwarps),
415  ceildiv(num_elems, warp_size_));
416  if (bucket < nwarps) {
417  srow[bucket]++;
418  }
419  }
420  // find starting row for thread i
421  for (size_type i = 1; i < nwarps; i++) {
422  srow[i] += srow[i - 1];
423  }
424  if (!is_srow_on_host) {
425  *mtx_srow = srow_host;
426  }
427  }
428  }

430  int64_t clac_size(const int64_t nnz) override
431  {
432  if (warp_size_ > 0) {
433  int multiple = 8;
434  if (nnz >= 2e6) {
435  multiple = 128;
436  } else if (nnz >= 2e5) {
437  multiple = 32;
438  }
439 
440 #if GINKGO_HIP_PLATFORM_HCC
441  if (!cuda_strategy_) {
442  multiple = 8;
443  if (nnz >= 1e7) {
444  multiple = 64;
445  } else if (nnz >= 1e6) {
446  multiple = 16;
447  }
448  }
449 #endif // GINKGO_HIP_PLATFORM_HCC
450 
451  auto nwarps = nwarps_ * multiple;
452  return min(ceildiv(nnz, warp_size_), int64_t(nwarps));
453  } else {
454  return 0;
455  }
456  }

470  class automatical : public strategy_type {
471  public:
472  /* Use imbalance strategy when the maximum number of nonzero per row is

478  /* Use imbalance strategy when the maximum number of nonzero per row is

531  void process(const Array<index_type> &mtx_row_ptrs,
532  Array<index_type> *mtx_srow) override
533  {
534  // if the number of stored elements is larger than <nnz_limit> or
535  // the maximum number of stored elements per row is larger than
536  // <row_len_limit>, use load_balance otherwise use classical
537  index_type nnz_limit = nvidia_nnz_limit;
538  index_type row_len_limit = nvidia_row_len_limit;
539 #if GINKGO_HIP_PLATFORM_HCC
540  if (!cuda_strategy_) {
541  nnz_limit = amd_nnz_limit;
542  row_len_limit = amd_row_len_limit;
543  }
544 #endif // GINKGO_HIP_PLATFORM_HCC
545  auto host_mtx_exec = mtx_row_ptrs.get_executor()->get_master();
546  const bool is_mtx_on_host{host_mtx_exec ==
547  mtx_row_ptrs.get_executor()};
548  Array<index_type> row_ptrs_host(host_mtx_exec);
549  const index_type *row_ptrs{};
550  if (is_mtx_on_host) {
551  row_ptrs = mtx_row_ptrs.get_const_data();
552  } else {
553  row_ptrs_host = mtx_row_ptrs;
554  row_ptrs = row_ptrs_host.get_const_data();
555  }
556  const auto num_rows = mtx_row_ptrs.get_num_elems() - 1;
557  if (row_ptrs[num_rows] > nnz_limit) {
558  load_balance actual_strategy(nwarps_, warp_size_,
559  cuda_strategy_);
560  if (is_mtx_on_host) {
561  actual_strategy.process(mtx_row_ptrs, mtx_srow);
562  } else {
563  actual_strategy.process(row_ptrs_host, mtx_srow);
564  }
565  this->set_name(actual_strategy.get_name());
566  } else {
567  index_type maxnum = 0;
568  for (index_type i = 1; i < num_rows + 1; i++) {
569  maxnum = max(maxnum, row_ptrs[i] - row_ptrs[i - 1]);
570  }
571  if (maxnum > row_len_limit) {
572  load_balance actual_strategy(nwarps_, warp_size_,
573  cuda_strategy_);
574  if (is_mtx_on_host) {
575  actual_strategy.process(mtx_row_ptrs, mtx_srow);
576  } else {
577  actual_strategy.process(row_ptrs_host, mtx_srow);
578  }
579  this->set_name(actual_strategy.get_name());
580  } else {
581  classical actual_strategy;
582  if (is_mtx_on_host) {
583  actual_strategy.process(mtx_row_ptrs, mtx_srow);
584  max_length_per_row_ =
585  actual_strategy.get_max_length_per_row();
586  } else {
587  actual_strategy.process(row_ptrs_host, mtx_srow);
588  max_length_per_row_ =
589  actual_strategy.get_max_length_per_row();
590  }
591  this->set_name(actual_strategy.get_name());
592  }
593  }
594  }

901  void convert_strategy_helper(CsrType *result) const
902  {
903  auto strat = this->get_strategy().get();
904  std::shared_ptr<typename CsrType::strategy_type> new_strat;
905  if (dynamic_cast<classical *>(strat)) {
906  new_strat = std::make_shared<typename CsrType::classical>();
907  } else if (dynamic_cast<merge_path *>(strat)) {
908  new_strat = std::make_shared<typename CsrType::merge_path>();
909  } else if (dynamic_cast<cusparse *>(strat)) {
910  new_strat = std::make_shared<typename CsrType::cusparse>();
911  } else if (dynamic_cast<sparselib *>(strat)) {
912  new_strat = std::make_shared<typename CsrType::sparselib>();
913  } else {
914  auto rexec = result->get_executor();
915  auto cuda_exec =
916  std::dynamic_pointer_cast<const CudaExecutor>(rexec);
917  auto hip_exec = std::dynamic_pointer_cast<const HipExecutor>(rexec);
918  auto lb = dynamic_cast<load_balance *>(strat);
919  if (cuda_exec) {
920  if (lb) {
921  new_strat =
922  std::make_shared<typename CsrType::load_balance>(
923  cuda_exec);
924  } else {
925  new_strat = std::make_shared<typename CsrType::automatical>(
926  cuda_exec);
927  }
928  } else if (hip_exec) {
929  if (lb) {
930  new_strat =
931  std::make_shared<typename CsrType::load_balance>(
932  hip_exec);
933  } else {
934  new_strat = std::make_shared<typename CsrType::automatical>(
935  hip_exec);
936  }
937  } else {
938  // Try to preserve this executor's configuration
939  auto this_cuda_exec =
940  std::dynamic_pointer_cast<const CudaExecutor>(
941  this->get_executor());
942  auto this_hip_exec =
943  std::dynamic_pointer_cast<const HipExecutor>(
944  this->get_executor());
945  if (this_cuda_exec) {
946  if (lb) {
947  new_strat =
948  std::make_shared<typename CsrType::load_balance>(
949  this_cuda_exec);
950  } else {
951  new_strat =
952  std::make_shared<typename CsrType::automatical>(
953  this_cuda_exec);
954  }
955  } else if (this_hip_exec) {
956  if (lb) {
957  new_strat =
958  std::make_shared<typename CsrType::load_balance>(
959  this_hip_exec);
960  } else {
961  new_strat =
962  std::make_shared<typename CsrType::automatical>(
963  this_hip_exec);
964  }
965  } else {
966  // We had a load balance or automatical strategy from a non
967  // HIP or Cuda executor and are moving to a non HIP or Cuda
968  // executor.
969  // FIXME this creates a long delay
970  if (lb) {
971  new_strat =
972  std::make_shared<typename CsrType::load_balance>();
973  } else {
974  new_strat =
975  std::make_shared<typename CsrType::automatical>();
976  }
977  }
978  }
979  }
980  result->set_strategy(new_strat);
981  }
