329  void process(const Array<index_type> &mtx_row_ptrs,
330  Array<index_type> *mtx_srow) override
331  {
332  auto nwarps = mtx_srow->get_num_elems();
333 
334  if (nwarps > 0) {
335  auto host_srow_exec = mtx_srow->get_executor()->get_master();
336  auto host_mtx_exec = mtx_row_ptrs.get_executor()->get_master();
337  const bool is_srow_on_host{host_srow_exec ==
338  mtx_srow->get_executor()};
339  const bool is_mtx_on_host{host_mtx_exec ==
340  mtx_row_ptrs.get_executor()};
341  Array<index_type> row_ptrs_host(host_mtx_exec);
342  Array<index_type> srow_host(host_srow_exec);
343  const index_type *row_ptrs{};
344  index_type *srow{};
345  if (is_srow_on_host) {
346  srow = mtx_srow->get_data();
347  } else {
348  srow_host = *mtx_srow;
349  srow = srow_host.get_data();
350  }
351  if (is_mtx_on_host) {
352  row_ptrs = mtx_row_ptrs.get_const_data();
353  } else {
354  row_ptrs_host = mtx_row_ptrs;
355  row_ptrs = row_ptrs_host.get_const_data();
356  }
357  for (size_type i = 0; i < nwarps; i++) {
358  srow[i] = 0;
359  }
360  const auto num_rows = mtx_row_ptrs.get_num_elems() - 1;
361  const auto num_elems = row_ptrs[num_rows];
362  for (size_type i = 0; i < num_rows; i++) {
363  auto bucket =
364  ceildiv((ceildiv(row_ptrs[i + 1], warp_size_) * nwarps),
365  ceildiv(num_elems, warp_size_));
366  if (bucket < nwarps) {
367  srow[bucket]++;
368  }
369  }
370  // find starting row for thread i
371  for (size_type i = 1; i < nwarps; i++) {
372  srow[i] += srow[i - 1];
373  }
374  if (!is_srow_on_host) {
375  *mtx_srow = srow_host;
376  }
377  }
378  }

462  void process(const Array<index_type> &mtx_row_ptrs,
463  Array<index_type> *mtx_srow) override
464  {
465  // if the number of stored elements is larger than <nnz_limit> or
466  // the maximum number of stored elements per row is larger than
467  // <row_len_limit>, use load_balance otherwise use classical
468  // CUDA: nnz_limit = 1e6, row_len_limit = 1024
469  // AMD: nnz_limit = 1e8, row_len_limit = 768
470  index_type nnz_limit = 1e6;
471  index_type row_len_limit = 1024;
472 #if GINKGO_HIP_PLATFORM_HCC
473  if (!cuda_strategy_) {
474  nnz_limit = 1e8;
475  row_len_limit = 768;
476  }
477 #endif // GINKGO_HIP_PLATFORM_HCC
478  auto host_mtx_exec = mtx_row_ptrs.get_executor()->get_master();
479  const bool is_mtx_on_host{host_mtx_exec ==
480  mtx_row_ptrs.get_executor()};
481  Array<index_type> row_ptrs_host(host_mtx_exec);
482  const index_type *row_ptrs{};
483  if (is_mtx_on_host) {
484  row_ptrs = mtx_row_ptrs.get_const_data();
485  } else {
486  row_ptrs_host = mtx_row_ptrs;
487  row_ptrs = row_ptrs_host.get_const_data();
488  }
489  const auto num_rows = mtx_row_ptrs.get_num_elems() - 1;
490  if (row_ptrs[num_rows] > nnz_limit) {
491  load_balance actual_strategy(nwarps_, warp_size_,
492  cuda_strategy_);
493  if (is_mtx_on_host) {
494  actual_strategy.process(mtx_row_ptrs, mtx_srow);
495  } else {
496  actual_strategy.process(row_ptrs_host, mtx_srow);
497  }
498  this->set_name(actual_strategy.get_name());
499  } else {
500  index_type maxnum = 0;
501  for (index_type i = 1; i < num_rows + 1; i++) {
502  maxnum = max(maxnum, row_ptrs[i] - row_ptrs[i - 1]);
503  }
504  if (maxnum > row_len_limit) {
505  load_balance actual_strategy(nwarps_, warp_size_,
506  cuda_strategy_);
507  if (is_mtx_on_host) {
508  actual_strategy.process(mtx_row_ptrs, mtx_srow);
509  } else {
510  actual_strategy.process(row_ptrs_host, mtx_srow);
511  }
512  this->set_name(actual_strategy.get_name());
513  } else {
514  classical actual_strategy;
515  if (is_mtx_on_host) {
516  actual_strategy.process(mtx_row_ptrs, mtx_srow);
517  max_length_per_row_ =
518  actual_strategy.get_max_length_per_row();
519  } else {
520  actual_strategy.process(row_ptrs_host, mtx_srow);
521  max_length_per_row_ =
522  actual_strategy.get_max_length_per_row();
523  }
524  this->set_name(actual_strategy.get_name());
525  }
526  }

840 template <typename ValueType, typename IndexType>
841 void strategy_rebuild_helper(Csr<ValueType, IndexType> *result)
842 {
843  using load_balance = typename Csr<ValueType, IndexType>::load_balance;
844  using automatical = typename Csr<ValueType, IndexType>::automatical;
845  auto strategy = result->get_strategy();
846  auto executor = result->get_executor();
847  if (std::dynamic_pointer_cast<load_balance>(strategy)) {
848  if (auto exec =
849  std::dynamic_pointer_cast<const HipExecutor>(executor)) {
850  result->set_strategy(std::make_shared<load_balance>(exec));
851  } else if (auto exec = std::dynamic_pointer_cast<const CudaExecutor>(
852  executor)) {
853  result->set_strategy(std::make_shared<load_balance>(exec));
854  }
855  } else if (std::dynamic_pointer_cast<automatical>(strategy)) {
856  if (auto exec =
857  std::dynamic_pointer_cast<const HipExecutor>(executor)) {
858  result->set_strategy(std::make_shared<automatical>(exec));
859  } else if (auto exec = std::dynamic_pointer_cast<const CudaExecutor>(
860  executor)) {
861  result->set_strategy(std::make_shared<automatical>(exec));
862  }
863  }
864 }
