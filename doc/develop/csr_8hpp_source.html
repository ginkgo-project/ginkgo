179  void process(const Array<index_type> &mtx_row_ptrs,
180  Array<index_type> *mtx_srow)
181  {
182  constexpr uint32 warp_size = 32;
183  auto nwarps = mtx_srow->get_num_elems();
184 
185  if (nwarps > 0) {
186  auto exec = mtx_srow->get_executor()->get_master();
187  Array<index_type> srow_host(exec);
188  srow_host = *mtx_srow;
189  auto srow = srow_host.get_data();
190  Array<index_type> row_ptrs_host(exec);
191  row_ptrs_host = mtx_row_ptrs;
192  auto row_ptrs = row_ptrs_host.get_const_data();
193  for (size_type i = 0; i < nwarps; i++) {
194  srow[i] = 0;
195  }
196  auto num_rows = mtx_row_ptrs.get_num_elems() - 1;
197  auto num_elems = row_ptrs[num_rows];
198  for (size_type i = 0; i < num_rows; i++) {
199  auto bucket =
200  ceildiv((ceildiv(row_ptrs[i + 1], warp_size) * nwarps),
201  ceildiv(num_elems, warp_size));
202  if (bucket < nwarps) {
203  srow[bucket]++;
204  }
205  }
206  // find starting row for thread i
207  for (size_type i = 1; i < nwarps; i++) {
208  srow[i] += srow[i - 1];
209  }
210  *mtx_srow = srow_host;
211  }
212  }

214  int64_t clac_size(const int64_t nnz)
215  {
216  constexpr uint32 warp_size = 32;
217  int multiple = 8;
218  if (nnz >= 2000000) {
219  multiple = 128;
220  } else if (nnz >= 200000) {
221  multiple = 32;
222  }
223  auto nwarps = nwarps_ * multiple;
224  return min(ceildiv(nnz, warp_size), static_cast<int64_t>(nwarps));
225  }

228  int64_t nwarps_;
229  };

246  void process(const Array<index_type> &mtx_row_ptrs,
247  Array<index_type> *mtx_srow)
248  {
249  // if the number of stored elements is larger than 1e6 or
250  // the maximum number of stored elements per row is larger than
251  // 64, use load_balance otherwise use classical
252  const auto num_rows = mtx_row_ptrs.get_num_elems() - 1;
253  Array<index_type> host_row_ptrs(
254  mtx_row_ptrs.get_executor()->get_master());
255  host_row_ptrs = mtx_row_ptrs;
256  const auto row_val = host_row_ptrs.get_const_data();
257  if (row_val[num_rows] > static_cast<index_type>(1e6)) {
258  std::make_shared<load_balance>(nwarps_)->process(host_row_ptrs,
259  mtx_srow);
260  this->set_name("load_balance");
261  } else {
262  index_type maxnum = 0;
263  for (index_type i = 1; i < num_rows + 1; i++) {
264  maxnum = max(maxnum, row_val[i] - row_val[i - 1]);
265  }
266  if (maxnum > 64) {
267  std::make_shared<load_balance>(nwarps_)->process(
268  host_row_ptrs, mtx_srow);
269  this->set_name("load_balance");
270  } else {
271  std::make_shared<classical>()->process(host_row_ptrs,
272  mtx_srow);
273  this->set_name("classical");
274  }
275  }
276  }
