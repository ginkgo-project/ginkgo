195  void apply_impl(const LinOp *in, LinOp *out) const
196  {
197  auto perm = as<Permutable<index_type>>(in);
198  std::unique_ptr<gko::LinOp> tmp{};
199  if (enabled_permute_ & inverse_permute) {
200  if (enabled_permute_ & row_permute) {
201  tmp = perm->inverse_row_permute(&permutation_);
202  }
203  if (enabled_permute_ & column_permute) {
204  if (enabled_permute_ & row_permute) {
205  tmp = as<Permutable<index_type>>(tmp.get())
206  ->inverse_column_permute(&permutation_);
207  } else {
208  tmp = perm->inverse_column_permute(&permutation_);
209  }
210  }
211  } else {
212  if (enabled_permute_ & row_permute) {
213  tmp = perm->row_permute(&permutation_);
214  }
215  if (enabled_permute_ & column_permute) {
216  if (enabled_permute_ & row_permute) {
217  tmp = as<Permutable<index_type>>(tmp.get())->column_permute(
218  &permutation_);
219  } else {
220  tmp = perm->column_permute(&permutation_);
221  }
222  }
223  }
224  out->copy_from(std::move(tmp));
225  }
