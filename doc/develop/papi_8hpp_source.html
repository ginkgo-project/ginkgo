236  size_type &get_counter(const PointerType *ptr)
237  {
238  const auto tmp = reinterpret_cast<uintptr>(ptr);
239  if (data.find(tmp) == data.end()) {
240  data[tmp] = 0;
241  }
242  auto &value = data[tmp];
243  if (!value) {
244  std::ostringstream oss;
245  oss << counter_name << "::" << tmp;
246  papi_sde_register_counter(*handle, oss.str().c_str(),
247  PAPI_SDE_RO | PAPI_SDE_INSTANT,
248  PAPI_SDE_long_long, &value);
249  }
250  return data[tmp];
251  }
