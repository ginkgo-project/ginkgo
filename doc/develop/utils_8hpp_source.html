418 template <typename T>
419 class temporary_clone {
420 public:
421  using value_type = T;
422  using pointer = T *;
423 
430  explicit temporary_clone(std::shared_ptr<const Executor> exec, pointer ptr)
431  {
432  if (ptr->get_executor() == exec) {
433  // just use the object we already have
434  handle_ = handle_type(ptr, [](pointer) {});
435  } else {
436  // clone the object to the new executor and make sure it's copied
437  // back before we delete it
438  handle_ = handle_type(gko::clone(std::move(exec), ptr).release(),
439  copy_back_deleter<T>(ptr));
440  }
441  }
442 
448  T *get() const { return handle_.get(); }
449 
450 private:
451  // std::function deleter allows to decide the (type of) deleter at runtime
452  using handle_type = std::unique_ptr<T, std::function<void(T *)>>;
453 
454  handle_type handle_;
455 };
