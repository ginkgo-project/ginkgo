426 template <typename T>
427 class temporary_clone {
428 public:
429  using value_type = T;
430  using pointer = T *;
431 
438  explicit temporary_clone(std::shared_ptr<const Executor> exec, pointer ptr)
439  {
440  if (ptr->get_executor() == exec) {
441  // just use the object we already have
442  handle_ = handle_type(ptr, [](pointer) {});
443  } else {
444  // clone the object to the new executor and make sure it's copied
445  // back before we delete it
446  handle_ = handle_type(gko::clone(std::move(exec), ptr).release(),
447  copy_back_deleter<T>(ptr));
448  }
449  }
450 
456  T *get() const { return handle_.get(); }
457 
458 private:
459  // std::function deleter allows to decide the (type of) deleter at runtime
460  using handle_type = std::unique_ptr<T, std::function<void(T *)>>;
461 
462  handle_type handle_;
463 };
