422 template <typename T>
423 class temporary_clone {
424 public:
425  using value_type = T;
426  using pointer = T *;
427 
434  explicit temporary_clone(std::shared_ptr<const Executor> exec, pointer ptr)
435  {
436  if (ptr->get_executor() == exec) {
437  // just use the object we already have
438  handle_ = handle_type(ptr, [](pointer) {});
439  } else {
440  // clone the object to the new executor and make sure it's copied
441  // back before we delete it
442  handle_ = handle_type(gko::clone(std::move(exec), ptr).release(),
443  copy_back_deleter<T>(ptr));
444  }
445  }
446 
452  T *get() const { return handle_.get(); }
453 
459  T *operator->() const { return handle_.get(); }
460 
461 private:
462  // std::function deleter allows to decide the (type of) deleter at runtime
463  using handle_type = std::unique_ptr<T, std::function<void(T *)>>;
464 
465  handle_type handle_;
466 };
