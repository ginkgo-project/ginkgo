504 template <typename T>
505 class temporary_clone {
506 public:
507  using value_type = T;
508  using pointer = T *;
509 
516  explicit temporary_clone(std::shared_ptr<const Executor> exec, pointer ptr)
517  {
518  if (ptr->get_executor() == exec) {
519  // just use the object we already have
520  handle_ = handle_type(ptr, [](pointer) {});
521  } else {
522  // clone the object to the new executor and make sure it's copied
523  // back before we delete it
524  handle_ = handle_type(gko::clone(std::move(exec), ptr).release(),
525  copy_back_deleter<T>(ptr));
526  }
527  }

534  T *get() const { return handle_.get(); }

541  T *operator->() const { return handle_.get(); }
542 
543 private:

545  using handle_type = std::unique_ptr<T, std::function<void(T *)>>;
