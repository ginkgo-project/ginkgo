253  size_type compute_ell_num_stored_elements_per_row(
254  Array<size_type> *row_nnz) const override
255  {
256  auto row_nnz_val = row_nnz->get_data();
257  auto num_rows = row_nnz->get_num_elems();
258  if (num_rows == 0) {
259  return 0;
260  }
261  std::sort(row_nnz_val, row_nnz_val + num_rows);
262  if (percent_ < 1) {
263  auto percent_pos = static_cast<size_type>(num_rows * percent_);
264  return row_nnz_val[percent_pos];
265  } else {
266  return row_nnz_val[num_rows - 1];
267  }
268  }

747 template <typename ValueType, typename IndexType>
748 template <typename HybType>
749 std::shared_ptr<typename HybType::strategy_type>
750 Hybrid<ValueType, IndexType>::get_strategy() const
751 {
752  static_assert(
753  std::is_same<HybType, Hybrid<typename HybType::value_type,
754  typename HybType::index_type>>::value,
755  "The given `HybType` type must be of type `matrix::Hybrid`!");
756 
757  std::shared_ptr<typename HybType::strategy_type> strategy;
758  if (std::dynamic_pointer_cast<automatic>(strategy_)) {
759  strategy = std::make_shared<typename HybType::automatic>();
760  } else if (auto temp = std::dynamic_pointer_cast<minimal_storage_limit>(
761  strategy_)) {
762  // minimal_storage_limit is related to ValueType and IndexType size.
763  if (sizeof(value_type) == sizeof(typename HybType::value_type) &&
764  sizeof(index_type) == sizeof(typename HybType::index_type)) {
765  strategy =
766  std::make_shared<typename HybType::minimal_storage_limit>();
767  } else {
768  strategy = std::make_shared<typename HybType::imbalance_limit>(
769  temp->get_percentage());
770  }
771  } else if (auto temp = std::dynamic_pointer_cast<imbalance_bounded_limit>(
772  strategy_)) {
773  strategy = std::make_shared<typename HybType::imbalance_bounded_limit>(
774  temp->get_percentage(), temp->get_ratio());
775  } else if (auto temp =
776  std::dynamic_pointer_cast<imbalance_limit>(strategy_)) {
777  strategy = std::make_shared<typename HybType::imbalance_limit>(
778  temp->get_percentage());
779  } else if (auto temp = std::dynamic_pointer_cast<column_limit>(strategy_)) {
780  strategy = std::make_shared<typename HybType::column_limit>(
781  temp->get_num_columns());
782  } else {
783  GKO_NOT_SUPPORTED(strategy_);
784  }
785  return strategy;
786 }
