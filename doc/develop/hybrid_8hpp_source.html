116  void compute_hybrid_config(const Array<size_type> &row_nnz,
117  size_type *ell_num_stored_elements_per_row,
118  size_type *coo_nnz)
119  {
120  Array<size_type> ref_row_nnz(row_nnz.get_executor()->get_master(),
121  row_nnz.get_num_elems());
122  ref_row_nnz = row_nnz;
123  ell_num_stored_elements_per_row_ =
124  this->compute_ell_num_stored_elements_per_row(&ref_row_nnz);
125  coo_nnz_ = this->compute_coo_nnz(ref_row_nnz);
126  *ell_num_stored_elements_per_row = ell_num_stored_elements_per_row_;
127  *coo_nnz = coo_nnz_;
128  }

166  size_type compute_coo_nnz(const Array<size_type> &row_nnz) const
167  {
168  size_type coo_nnz = 0;
169  auto row_nnz_val = row_nnz.get_const_data();
170  for (size_type i = 0; i < row_nnz.get_num_elems(); i++) {
171  if (row_nnz_val[i] > ell_num_stored_elements_per_row_) {
172  coo_nnz +=
173  row_nnz_val[i] - ell_num_stored_elements_per_row_;
174  }
175  }
176  return coo_nnz;
177  }
178 
179  private:
180  size_type ell_num_stored_elements_per_row_;
181  size_type coo_nnz_;

216  class imbalance_limit : public strategy_type {
217  public:
224  explicit imbalance_limit(float percent = 0.8) : percent_(percent)
225  {
226  percent_ = std::min(percent_, 1.0f);
227  percent_ = std::max(percent_, 0.0f);
228  }
229 
230  size_type compute_ell_num_stored_elements_per_row(
231  Array<size_type> *row_nnz) const override
232  {
233  auto row_nnz_val = row_nnz->get_data();
234  auto num_rows = row_nnz->get_num_elems();
235  std::sort(row_nnz_val, row_nnz_val + num_rows);
236  if (percent_ < 1) {
237  auto percent_pos = static_cast<size_type>(num_rows * percent_);
238  return row_nnz_val[percent_pos];
239  } else {
240  return row_nnz_val[num_rows - 1];
241  }
242  }
243 
244  private:
245  float percent_;
246  };
