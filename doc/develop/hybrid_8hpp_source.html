239  size_type compute_ell_num_stored_elements_per_row(
240  Array<size_type> *row_nnz) const override
241  {
242  auto row_nnz_val = row_nnz->get_data();
243  auto num_rows = row_nnz->get_num_elems();
244  std::sort(row_nnz_val, row_nnz_val + num_rows);
245  if (percent_ < 1) {
246  auto percent_pos = static_cast<size_type>(num_rows * percent_);
247  return row_nnz_val[percent_pos];
248  } else {
249  return row_nnz_val[num_rows - 1];
250  }
251  }
252 
253  private:
254  float percent_;
255  };
