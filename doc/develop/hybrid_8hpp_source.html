237  size_type compute_ell_num_stored_elements_per_row(
238  Array<size_type> *row_nnz) const override
239  {
240  auto row_nnz_val = row_nnz->get_data();
241  auto num_rows = row_nnz->get_num_elems();
242  std::sort(row_nnz_val, row_nnz_val + num_rows);
243  if (percent_ < 1) {
244  auto percent_pos = static_cast<size_type>(num_rows * percent_);
245  return row_nnz_val[percent_pos];
246  } else {
247  return row_nnz_val[num_rows - 1];
248  }
249  }

251  private:
252  float percent_;
