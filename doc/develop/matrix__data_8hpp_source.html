345  template <typename RandomDistribution, typename RandomEngine>
346  static matrix_data cond(size_type size,
347  remove_complex<ValueType> condition_number,
348  RandomDistribution &&dist, RandomEngine &&engine,
349  size_type num_reflectors)
350  {
351  using range = range<accessor::row_major<ValueType, 2>>;
352  std::vector<ValueType> mtx_data(size * size, zero<ValueType>());
353  std::vector<ValueType> ref_data(size);
354  std::vector<ValueType> work(size);
355  range matrix(mtx_data.data(), size, size, size);
356  range reflector(ref_data.data(), size, 1u, 1u);
357 
358  initialize_diag_with_cond(condition_number, matrix);
359  for (size_type i = 0; i < num_reflectors; ++i) {
360  generate_random_reflector(dist, engine, reflector);
361  reflect_domain(reflector, matrix, work.data());
362  generate_random_reflector(dist, engine, reflector);
363  reflect_range(reflector, matrix, work.data());
364  }
365  return matrix;
366  }

425  template <typename Accessor>
426  static void initialize_diag_with_cond(
427  remove_complex<ValueType> condition_number,
428  const range<Accessor> &matrix)
429  {
430  using sigma_type = remove_complex<ValueType>;
431  const auto size = matrix.length(0);
432  const auto min_sigma = one(condition_number) / sqrt(condition_number);
433  const auto max_sigma = sqrt(condition_number);
434 
435  matrix = zero(matrix);
436  for (gko::size_type i = 0; i < size; ++i) {
437  matrix(i, i) = max_sigma * static_cast<sigma_type>(size - i - 1) /
438  static_cast<sigma_type>(size - 1) +
439  min_sigma * static_cast<sigma_type>(i) /
440  static_cast<sigma_type>(size - 1);
441  }
442  }

455  template <typename Accessor>
456  static void reflect_domain(const range<Accessor> &reflector,
457  const range<Accessor> &matrix,
458  ValueType *work_data)
459  {
460  const auto two = one<ValueType>() + one<ValueType>();
461  range<accessor::row_major<ValueType, 2>> work(work_data,
462  matrix.length(0), 1u, 1u);
463  work = mmul(matrix, reflector);
464  const auto ct_reflector = conj(transpose(reflector));
465  const auto scale = two / mmul(ct_reflector, reflector)(0, 0);
466  matrix = matrix - scale * mmul(work, ct_reflector);
467  }
468 
469  template <typename Accessor>
470  static void reflect_range(const range<Accessor> &reflector,
471  const range<Accessor> &matrix,
472  ValueType *work_data)
473  {
474  const auto two = one<ValueType>() + one<ValueType>();
475  range<accessor::row_major<ValueType, 2>> work(
476  work_data, 1u, matrix.length(0), matrix.length(0));
477  const auto ct_reflector = conj(transpose(reflector));
478  work = mmul(ct_reflector, matrix);
479  const auto scale = two / mmul(ct_reflector, reflector)(0, 0);
480  matrix = matrix - scale * mmul(reflector, work);
481  }
