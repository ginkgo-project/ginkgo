179  void process(const Array<index_type> &mtx_row_ptrs,
180  Array<index_type> *mtx_srow)
181  {
182  constexpr uint32 warp_size = 32;
183  auto nwarps = mtx_srow->get_num_elems();
184 
185  if (nwarps > 0) {
186  auto host_srow_exec = mtx_srow->get_executor()->get_master();
187  auto host_mtx_exec = mtx_row_ptrs.get_executor()->get_master();
188  const bool is_srow_on_host{host_srow_exec ==
189  mtx_srow->get_executor()};
190  const bool is_mtx_on_host{host_mtx_exec ==
191  mtx_row_ptrs.get_executor()};
192  Array<index_type> row_ptrs_host(host_mtx_exec);
193  Array<index_type> srow_host(host_srow_exec);
194  const index_type *row_ptrs{};
195  index_type *srow{};
196  if (is_srow_on_host) {
197  srow = mtx_srow->get_data();
198  } else {
199  srow_host = *mtx_srow;
200  srow = srow_host.get_data();
201  }
202  if (is_mtx_on_host) {
203  row_ptrs = mtx_row_ptrs.get_const_data();
204  } else {
205  row_ptrs_host = mtx_row_ptrs;
206  row_ptrs = row_ptrs_host.get_const_data();
207  }
208  for (size_type i = 0; i < nwarps; i++) {
209  srow[i] = 0;
210  }
211  const auto num_rows = mtx_row_ptrs.get_num_elems() - 1;
212  const auto num_elems = row_ptrs[num_rows];
213  for (size_type i = 0; i < num_rows; i++) {
214  auto bucket =
215  ceildiv((ceildiv(row_ptrs[i + 1], warp_size) * nwarps),
216  ceildiv(num_elems, warp_size));
217  if (bucket < nwarps) {
218  srow[bucket]++;
219  }
220  }
221  // find starting row for thread i
222  for (size_type i = 1; i < nwarps; i++) {
223  srow[i] += srow[i - 1];
224  }
225  if (!is_srow_on_host) {
226  *mtx_srow = srow_host;
227  }
228  }
229  }

231  int64_t clac_size(const int64_t nnz)
232  {
233  constexpr uint32 warp_size = 32;
234  int multiple = 8;
235  if (nnz >= 2000000) {
236  multiple = 128;
237  } else if (nnz >= 200000) {
238  multiple = 32;
239  }
240  auto nwarps = nwarps_ * multiple;
241  return min(ceildiv(nnz, warp_size), static_cast<int64_t>(nwarps));
242  }

263  void process(const Array<index_type> &mtx_row_ptrs,
264  Array<index_type> *mtx_srow)
265  {
266  // if the number of stored elements is larger than 1e6 or
267  // the maximum number of stored elements per row is larger than
268  // 64, use load_balance otherwise use classical
269  auto host_mtx_exec = mtx_row_ptrs.get_executor()->get_master();
270  const bool is_mtx_on_host{host_mtx_exec ==
271  mtx_row_ptrs.get_executor()};
272  Array<index_type> row_ptrs_host(host_mtx_exec);
273  const index_type *row_ptrs{};
274  if (is_mtx_on_host) {
275  row_ptrs = mtx_row_ptrs.get_const_data();
276  } else {
277  row_ptrs_host = mtx_row_ptrs;
278  row_ptrs = row_ptrs_host.get_const_data();
279  }
280  const auto num_rows = mtx_row_ptrs.get_num_elems() - 1;
281  if (row_ptrs[num_rows] > index_type(1e6)) {
282  load_balance actual_strategy(nwarps_);
283  if (is_mtx_on_host) {
284  actual_strategy.process(mtx_row_ptrs, mtx_srow);
285  } else {
286  actual_strategy.process(row_ptrs_host, mtx_srow);
287  }
288  this->set_name(actual_strategy.get_name());
289  } else {
290  index_type maxnum = 0;
291  for (index_type i = 1; i < num_rows + 1; i++) {
292  maxnum = max(maxnum, row_ptrs[i] - row_ptrs[i - 1]);
293  }
294  if (maxnum > 64) {
295  load_balance actual_strategy(nwarps_);
296  if (is_mtx_on_host) {
297  actual_strategy.process(mtx_row_ptrs, mtx_srow);
298  } else {
299  actual_strategy.process(row_ptrs_host, mtx_srow);
300  }
301  this->set_name(actual_strategy.get_name());
302  } else {
303  classical actual_strategy;
304  if (is_mtx_on_host) {
305  actual_strategy.process(mtx_row_ptrs, mtx_srow);
306  } else {
307  actual_strategy.process(row_ptrs_host, mtx_srow);
308  }
309  this->set_name(actual_strategy.get_name());
310  }
311  }
312  }
