357  std::unique_ptr<Dense> create_submatrix(const span &rows,
358  const span &columns,
359  const size_type stride)
360  {
361  row_major_range range_this{this->get_values(), this->get_size()[0],

520 template <typename Matrix, typename... TArgs>
521 std::unique_ptr<Matrix> initialize(
522  size_type stride, std::initializer_list<typename Matrix::value_type> vals,
523  std::shared_ptr<const Executor> exec, TArgs &&... create_args)
524 {
525  using dense = matrix::Dense<typename Matrix::value_type>;
526  size_type num_rows = vals.size();
527  auto tmp = dense::create(exec->get_master(), dim<2>{num_rows, 1}, stride);
528  size_type idx = 0;
529  for (const auto &elem : vals) {
530  tmp->at(idx) = elem;
531  ++idx;
532  }
533  auto mtx = Matrix::create(exec, std::forward<TArgs>(create_args)...);
534  tmp->move_to(mtx.get());
535  return mtx;
536 }

590 template <typename Matrix, typename... TArgs>
591 std::unique_ptr<Matrix> initialize(
592  size_type stride,
593  std::initializer_list<std::initializer_list<typename Matrix::value_type>>
594  vals,
595  std::shared_ptr<const Executor> exec, TArgs &&... create_args)
596 {
597  using dense = matrix::Dense<typename Matrix::value_type>;
598  size_type num_rows = vals.size();
599  size_type num_cols = num_rows > 0 ? begin(vals)->size() : 1;
600  auto tmp =
601  dense::create(exec->get_master(), dim<2>{num_rows, num_cols}, stride);
602  size_type ridx = 0;
603  for (const auto &row : vals) {
604  size_type cidx = 0;
605  for (const auto &elem : row) {
606  tmp->at(ridx, cidx) = elem;
607  ++cidx;
608  }
609  ++ridx;
610  }
611  auto mtx = Matrix::create(exec, std::forward<TArgs>(create_args)...);
612  tmp->move_to(mtx.get());
613  return mtx;
614 }
