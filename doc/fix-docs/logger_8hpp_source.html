516 template <typename ConcreteLoggable, typename PolymorphicBase = Loggable>
517 class EnableLogging : public Loggable {
518 public:
519  void add_logger(std::shared_ptr<const Logger> logger) override
520  {
521  loggers_.push_back(logger);
522  }
523 
524  void remove_logger(const Logger *logger) override
525  {
526  auto idx = find_if(begin(loggers_), end(loggers_),
527  [&logger](std::shared_ptr<const Logger> l) {
528  return lend(l) == logger;
529  });
530  if (idx != end(loggers_)) {
531  loggers_.erase(idx);
532  } else {
533  throw OutOfBoundsError(__FILE__, __LINE__, loggers_.size(),
534  loggers_.size());
535  }
536  }
537 
538 protected:
539  template <size_type Event, typename... Params>
540  void log(Params &&... params) const
541  {
542  for (auto &logger : loggers_) {
543  logger->template on<Event>(std::forward<Params>(params)...);
544  }
545  }
546 
547  std::vector<std::shared_ptr<const Logger>> loggers_;
548 };
