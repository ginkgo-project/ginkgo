523 template <typename ConcreteLoggable, typename PolymorphicBase = Loggable>
524 class EnableLogging : public Loggable {
525 public:
526  void add_logger(std::shared_ptr<const Logger> logger) override
527  {
528  loggers_.push_back(logger);
529  }
530 
531  void remove_logger(const Logger *logger) override
532  {
533  auto idx = find_if(begin(loggers_), end(loggers_),
534  [&logger](std::shared_ptr<const Logger> l) {
535  return lend(l) == logger;
536  });
537  if (idx != end(loggers_)) {
538  loggers_.erase(idx);
539  } else {
540  throw OutOfBoundsError(__FILE__, __LINE__, loggers_.size(),
541  loggers_.size());
542  }
543  }
544 
545 protected:
546  template <size_type Event, typename... Params>
547  void log(Params &&... params) const
548  {
549  for (auto &logger : loggers_) {
550  logger->template on<Event>(std::forward<Params>(params)...);
551  }
552  }
553 
554  std::vector<std::shared_ptr<const Logger>> loggers_;
555 };
